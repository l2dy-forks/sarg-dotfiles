(defun exwm/switch-to-next-window (class)
  (cl-loop
   with first with this
   for b being the buffers
   for is-class = (string= class (buffer-local-value 'exwm-class-name b))
   when is-class do
   (setf first (or first b))
   (setf next (and this b))
   (setf this (or this (when (eq (current-buffer) b) b)))
   until next
   finally (switch-to-buffer (or next first) t t)))

  (if (or (= 0 (mod
                (calendar-day-of-week (calendar-current-date))

          (> (nth 2 (decode-time)) 18))
      (apply orig-fun args)

  "Asks to enter random string as confirmation before executing PROC."

                       (random (expt 16 6))
                       (random (expt 16 6))
                       (random (expt 16 6))))

         (input (read-string (format "Enter %s if you wish to proceed: " pass))))

    (if (string= input pass)
        (apply proc args))))
